		return ""
	}
	String prefix = strs.first;

	for(var i = 1; i < strs.length; i++){
		String str = strs[i];

			prefix = prefix.substring(0, prefix.length - 1);
			if(prefix.isEmpty){
				return "";
			}
		}
	}
	return prefix;

}
